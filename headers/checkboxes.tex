\newcommand{\checkbox}[1]{%
  {%
    \CheckBox[
      name=item#1,
      width=0.6em,
      height=0.6em,
      bordercolor={0 0 0}
    ]{}
  }%
}

% Read-only, checked checkbox
\newcommand{\checkboxlocked}[1]{%
  {%
    \CheckBox[
      name=lockeditem#1,
      width=0.6em,
      height=0.6em,
      bordercolor={0 0 0},
      checked,
      readonly
    ]{}
  }%
}
